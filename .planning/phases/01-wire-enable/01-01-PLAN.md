# Attach fee path hints for cross-all opportunities
for opp in opportunities:
    pa = opp.get("_platform_a", "")
    pb = opp.get("_platform_b", "")
    if not pa or not pb:
        continue
    prices_str = opp.get("prices", "")
    a_yes = a_no = b_yes = b_no = 0.0
    for part in prices_str.split():
        if "=" not in part:
            continue
        key, val = part.split("=", 1)
        v = float(val)
        if key.endswith("_Y"):
            if key.startswith(pa[:2].upper()):
                a_yes = v
            else:
                b_yes = v
        elif key.endswith("_N"):
            if key.startswith(pa[:2].upper()):
                a_no = v
            else:
                b_no = v
    platforms = [pa, pb]
    yes_p = {pa: a_yes, pb: b_yes} if a_yes and b_yes else {}
    no_p = {pa: a_no, pb: b_no} if a_no and b_no else {}
    if yes_p and no_p:
        fee_path = find_lowest_fee_path(platforms, yes_p, no_p)
        if fee_path:
            opp["_fee_path"] = fee_path

elif opp_type.startswith("Cross"):
    # Re-validate fee path if scan provided a hint (per user decision: confirm or override)
    fee_path = opportunity.get("_fee_path")
    if fee_path:
        fresh = find_lowest_fee_path(
            [fee_path["best_yes_platform"], fee_path["best_no_platform"]],
            {fee_path["best_yes_platform"]: fee_path["yes_price"]},
            {fee_path["best_no_platform"]: fee_path["no_price"]},
        )
        if fresh and fresh["net_profit"] > 0:
            logger.info(
                "Fee path re-validated: %s YES + %s NO, net=$%.4f",
                fresh["best_yes_platform"], fresh["best_no_platform"],
                fresh["net_profit"],
            )
            # USE the fee path result to build legs with optimal platform routing
            yes_plat = fresh["best_yes_platform"]
            no_plat = fresh["best_no_platform"]
            yes_price = fresh["yes_price"]
            no_price = fresh["no_price"]

            # Build YES leg
            yes_leg = {"platform": yes_plat, "side": "BUY", "token": "yes",
                       "price": yes_price}
            if yes_plat == "polymarket":
                yes_leg["_token_id"] = token_ids[0] if len(token_ids) > 0 else ""
            elif yes_plat == "kalshi":
                yes_leg["side"] = "yes"
                yes_leg["action"] = "buy"
                yes_leg["_ticker"] = opportunity.get("_kalshi_ticker", "")

            # Build NO leg
            no_leg = {"platform": no_plat, "side": "BUY", "token": "no",
                      "price": no_price}
            if no_plat == "polymarket":
                no_leg["_token_id"] = token_ids[1] if len(token_ids) > 1 else ""
            elif no_plat == "kalshi":
                no_leg["side"] = "no"
                no_leg["action"] = "buy"
                no_leg["_ticker"] = opportunity.get("_kalshi_ticker", "")

            legs = [yes_leg, no_leg]
            return legs
        else:
            logger.info("Fee path stale (no longer profitable), falling back to default routing")
            # Fall through to default prices_str routing below

    # Default routing: parse prices_str (unchanged existing logic)
    prices_str = opportunity.get("prices", "")
    kalshi_ticker = opportunity.get("_kalshi_ticker", "")

The key changes vs the previous advisory-only approach:
- When fresh fee path is valid (net_profit > 0), legs are built using `fresh["best_yes_platform"]` and `fresh["best_no_platform"]` for platform selection
- The YES and NO leg platforms come from the fee path, not from hardcoded PM/K parsing
- When fee path re-validation fails (returns None or net_profit <= 0), falls back to default prices_str routing
- When no _fee_path key exists at all, default routing is used (backward compatible)

For `_platform_a`-based cross-all opps (the `elif "_platform_a" in opportunity:` branch), the fee path routing is handled by `_build_cross_all_legs` which already parses platform dynamically — no change needed there. The fee path routing above handles the PM+K specific branches (lines 1054-1077).
