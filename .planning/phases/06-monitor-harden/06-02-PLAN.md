   ```javascript
   function loadLeaderboard() {
     fetch('/api/strategy-leaderboard')
       .then(r => {
         if (!r.ok) throw new Error('HTTP ' + r.status);
         return r.json();
       })
       .then(data => {
         const tbody = document.getElementById('leaderboard-body');
         const table = document.getElementById('leaderboard-table');
         const loading = document.getElementById('leaderboard-loading');
         const empty = document.getElementById('leaderboard-empty');
         const timestamp = document.getElementById('leaderboard-timestamp');
         
         tbody.innerHTML = '';
         
         if (!data.strategies || data.strategies.length === 0) {
           empty.style.display = 'block';
           table.style.display = 'none';
           loading.style.display = 'none';
           return;
         }
         
         data.strategies.forEach(strategy => {
           const row = document.createElement('tr');
           
           const cells = [
             strategy.strategy || 'unknown',
             String(strategy.trade_count || 0),
             String(strategy.wins || 0),
             formatPercent(strategy.win_rate),
             formatCurrency(strategy.total_pnl),
             formatCurrency(strategy.avg_pnl),
             formatSharpe(strategy.annual_sharpe),
             formatCurrency(strategy.max_drawdown)
           ];
           
           cells.forEach(cellText => {
             const cell = document.createElement('td');
             cell.textContent = cellText;
             row.appendChild(cell);
           });
           
           tbody.appendChild(row);
         });
         
         table.style.display = 'table';
         loading.style.display = 'none';
         empty.style.display = 'none';
         
         // Update timestamp
         const lastUpdated = new Date(data.timestamp * 1000).toLocaleTimeString();
         timestamp.textContent = 'Last updated: ' + lastUpdated;
       })
       .catch(err => {
         console.error('Leaderboard fetch failed:', err);
         document.getElementById('leaderboard-loading').textContent = 'Error loading metrics';
       });
   }
   
   function formatPercent(val) {
     return val === null || val === undefined ? 'N/A' : (val * 100).toFixed(1) + '%';
   }
   
   function formatCurrency(val) {
     return val === null || val === undefined ? 'N/A' : '$' + parseFloat(val).toFixed(4);
   }
   
   function formatSharpe(val) {
     return val === null || val === undefined ? 'N/A' : parseFloat(val).toFixed(3);
   }
   
   // Load on page load
   if (document.readyState === 'loading') {
     document.addEventListener('DOMContentLoaded', loadLeaderboard);
   } else {
     loadLeaderboard();
   }
   
   // Refresh every 30 seconds
   setInterval(loadLeaderboard, 30000);
   ```

5. Ensure the section has proper HTML structure and doesn't break existing page layout

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Analytics → Dashboard | Dashboard state receives pre-computed metrics from analytics; trusts structure matches expected fields |
| JavaScript → DOM | Dashboard UI populates table using textContent (safe) to prevent XSS injection of market names or strategy types |
| HTTP endpoint | `/api/strategy-leaderboard` returns JSON; consumer (browser) validates structure before rendering |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-05 | Spoofing | Leaderboard metrics reflect false P&L | mitigate | Metrics computed from trades.db which is write-only by executor (no direct user input); trust DB as source of truth |
| T-06-06 | Denial of Service | Large leaderboard response floods client | mitigate | Limit to top 100 strategies in response; paginate if needed. Add Content-Length header to prevent slowloris |
| T-06-07 | Elevation of Privilege | Unauthorized access to leaderboard | mitigate | Apply existing dashboard auth (_check_auth) to /api/strategy-leaderboard endpoint |
| T-06-08 | Information Disclosure | Strategy names in leaderboard expose market types | mitigate | Strategy names are not secrets (e.g., "binary", "cross", "kalshi"); acceptable to display publicly (if dashboard is public) |
| T-06-09 | Tampering | Strategy metrics manually edited in trades.db | mitigate | trades.db is SQLite; read-only DuckDB connection prevents tampering during analytics query |

</threat_model>
