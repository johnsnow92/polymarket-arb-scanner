elif opp_type == "PolymarketRewards":
    # Reward opportunities: resting limit orders on reward-eligible markets
    optimal_bid = opp.get("optimal_bid")
    optimal_ask = opp.get("optimal_ask")
    size = opp.get("size", 5.0)
    
    # Determine sides based on midpoint range (Polymarket reward rule)
    mid = (optimal_bid + optimal_ask) / 2
    sides = []
    if 0.10 <= mid <= 0.90:
        # Single-sided OK; post both for better reward score
        sides = [
            {"platform": "polymarket", "side": "BID", "price": optimal_bid, "size": size},
            {"platform": "polymarket", "side": "ASK", "price": optimal_ask, "size": size},
        ]
    else:
        # Outside range: must post both sides
        sides = [
            {"platform": "polymarket", "side": "BID", "price": optimal_bid, "size": size},
            {"platform": "polymarket", "side": "ASK", "price": optimal_ask, "size": size},
        ]
    
    return {
        "type": "reward_resting",
        "platform": "polymarket",
        "market_key": opp["_market_key"],
        "sides": sides,
        "size": size,
        "strategy": "rewards",
    }

elif leg_type == "reward_resting":
