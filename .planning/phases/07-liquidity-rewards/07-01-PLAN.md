```python
class RewardTracker:
    """Track Polymarket reward program metadata and calculate optimal reward-aware spreads.
    
    Polymarket Markets API returns incentives metadata per market:
    - min_incentive_size: minimum order size to qualify for rewards
    - max_incentive_spread: maximum spread to qualify for rewards (tighter = better score)
    - pool_size_usdc: total reward pool for this market
    - status: "active" or "inactive"
    
    Kalshi has no public API; local tracking via KalshiRewardTracker.
    """
    
    def __init__(self):
        self.reward_scores: dict[str, dict] = {}  # {market_key: {"polymarket": {...}, "kalshi": {...}}}
        self.last_update: dict[str, float] = {}  # {market_key: unix_timestamp}
        self._lock = threading.Lock()
    
    def update_polymarket_reward(self, market_key: str, reward_data: dict, ttl_seconds: float = 300.0) -> None:
        """Update reward metadata from Polymarket Markets API.
        
        Args:
            market_key: condition ID from Markets API
            reward_data: Dict with min_incentive_size, max_incentive_spread, pool_size_usdc, status
            ttl_seconds: Cache TTL (300s = 5 min, matches scan cycle)
        """
        with self._lock:
            if market_key not in self.reward_scores:
                self.reward_scores[market_key] = {}
            self.reward_scores[market_key]["polymarket"] = reward_data
            self.last_update[market_key] = time.time()
    
    def get_polymarket_reward(self, market_key: str) -> dict | None:
        """Get cached reward metadata for market, or None if stale/missing."""
        with self._lock:
            if market_key not in self.reward_scores:
                return None
            reward = self.reward_scores[market_key].get("polymarket")
            if not reward:
                return None
            # Check TTL (5 min cache)
            age = time.time() - self.last_update.get(market_key, 0)
            if age > 300.0:
                return None
            return reward
    
    def calculate_optimal_reward_spread(self, market_key: str, mid_price: float, 
                                       inventory: float = 0.0) -> dict | None:
        """Calculate bid/ask prices optimized for reward qualification.
        
        Returns:
            Dict with bid, ask, spread, reward_optimized (bool), or None if no reward data
        """
        reward = self.get_polymarket_reward(market_key)
        if not reward:
            return None
        
        max_spread = reward.get("max_incentive_spread", 0.05)
        
        # Target: 60% of max spread = higher reward score than competitors
        target_spread = max_spread * 0.6
        
        # Inventory skew: if long, skew toward selling (higher ask, lower bid)
        skew = 0.0
        if inventory > 0:
            skew = -target_spread * 0.1
        
        bid = mid_price - (target_spread / 2) + skew
        ask = mid_price + (target_spread / 2) + skew
        
        # Clamp to [0.01, 0.99] (Polymarket valid range)
        bid = max(0.01, min(0.99, bid))
        ask = max(0.01, min(0.99, ask))
        
        return {
            "bid": bid,
            "ask": ask,
            "spread": ask - bid,
            "reward_optimized": True,
        }


class KalshiRewardTracker:
    """Track Kalshi liquidity incentive program participation (no public API).
    
    Kalshi rewards resting limit orders; no public API for reward scores.
    This tracker logs placement and cancellation events to calculate resting time
    and spread metrics for manual reward estimation.
    """
    
    def __init__(self, db: 'TradeDB'):  # Assume TradeDB imported
        self.db = db
        self._active_orders: dict[str, dict] = {}  # order_id -> {market_key, placed_at, size, spread, side}
        self._lock = threading.Lock()
    
    def log_order_placed(self, order_id: str, market_key: str, size: float, 
                        price: float, mid_price: float, side: str) -> None:
        """Record Kalshi order placement for reward tracking."""
        spread = abs(price - mid_price) / mid_price if mid_price > 0 else 0
        
        with self._lock:
            self._active_orders[order_id] = {
                "market_key": market_key,
                "placed_at": time.time(),
                "size": size,
                "spread": spread,
                "side": side,
            }
        
        # Persist to DB
        self.db.log_reward_metric(
            platform="kalshi",
            market_key=market_key,
            order_id=order_id,
            event="placed",
            size=size,
            spread=spread,
            resting_seconds=0,
        )
    
    def log_order_cancelled(self, order_id: str) -> None:
        """Record order cancellation with final resting time."""
        with self._lock:
            if order_id not in self._active_orders:
                return
            order = self._active_orders.pop(order_id)
        
        resting_seconds = int(time.time() - order["placed_at"])
        
        self.db.log_reward_metric(
            platform="kalshi",
            market_key=order["market_key"],
            order_id=order_id,
            event="cancelled",
            size=order["size"],
            spread=order["spread"],
            resting_seconds=resting_seconds,
        )
    
    def get_active_orders(self) -> dict[str, dict]:
        """Get all currently active orders."""
        with self._lock:
            return dict(self._active_orders)
    
    def estimate_daily_reward(self, market_key: str) -> float:
        """Estimate daily reward for a market based on local resting metrics.
        
        Note: Actual Kalshi rewards computed daily at midnight UTC.
        This is an estimate only, for display/monitoring.
        """
        metrics = self.db._conn.execute("""
            SELECT SUM(resting_seconds), AVG(spread) FROM reward_metrics
            WHERE platform='kalshi' AND market_key=? AND event='cancelled'
        """, (market_key,)).fetchone()
        
        if not metrics or metrics[0] is None:
            return 0.0
        
        total_resting = metrics[0]
        avg_spread = metrics[1] or 0.05
        
        # Kalshi formula estimate: reward ∝ resting_time * spread_tightness
        # Assuming ~$0.50/day per 24h of resting at good spreads
        estimated_daily = (total_resting / 86400) * (1.0 - min(avg_spread, 0.20)) * 0.50
        return max(0.0, estimated_daily)
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Polymarket Markets API → RewardTracker | Untrusted reward metadata (min_size, max_spread, pool_size) could be malformed |
| Kalshi order data → KalshiRewardTracker | Order placement data from exchange is trusted (verified by nonce/auth) |
| Database → Reward metrics lookup | SQLite data is authoritative local source of truth |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-01 | Tampering | Polymarket reward metadata validation | mitigate | Validate all reward fields before use: check min_size > 0, 0 < max_spread <= 0.50, pool_size >= 0; reject markets with invalid data (log warning, skip) |
| T-07-02 | Denial of Service | RewardTracker cache exhaustion | accept | Single dict per market; bounded by # of Polymarket markets (~5k); no unbounded growth |
| T-07-03 | Information Disclosure | Kalshi order logs in database | accept | Logs contain only order metrics (size, spread, time) — no PII, no credentials, no account balances |
| T-07-04 | Tampering | Database reward_metrics table poisoning | accept | Database is local SQLite with WAL mode; file-level permissions sufficient. No multi-user access. |

</threat_model>
