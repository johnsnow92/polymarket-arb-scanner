```python
async def scan_rewards_loop():
    while running:
        try:
            if config.REWARDS_ENABLED:
                # Poll Polymarket reward metadata
                polymarket_reward_opps = scan_polymarket_rewards(
                    markets=all_polymarket_markets,
                    reward_tracker=reward_tracker,
                    price_cache=price_cache,
                )
                
                # Kalshi rewards
                kalshi_reward_opps = scan_kalshi_rewards(
                    kalshi_client=kalshi_client,
                    reward_tracker=kalshi_reward_tracker,
                )
                
                # Index opportunities for WS-triggered execution
                for opp in polymarket_reward_opps + kalshi_reward_opps:
                    opportunity_index.add(opp)
                
                logger.debug(f"Rewards scan: {len(polymarket_reward_opps)} Polymarket + {len(kalshi_reward_opps)} Kalshi")
        except Exception as e:
            logger.error(f"Rewards scanning error: {e}")
        
        await asyncio.sleep(config.REWARDS_POLL_INTERVAL)

# Add to event loop tasks
tasks.append(asyncio.create_task(scan_rewards_loop()))
```

```python
def estimate_reward_yield(reward_tracker) -> float:
    """Estimate daily reward yield from RewardTracker active markets."""
    if not reward_tracker:
        return 0.0
    
    total_yield = 0.0
    for market_key, reward_info in reward_tracker.reward_scores.items():
        poly_reward = reward_info.get("polymarket")
        if poly_reward:
            pool = poly_reward.get("pool_size_usdc", 0)
            # Rough estimate: daily yield = pool_size / 30 days
            daily_estimate = pool / 30
            total_yield += daily_estimate
    
    return total_yield

def calculate_total_exposure(reward_tracker) -> float:
    """Sum total capital deployed in reward resting orders."""
    if not reward_tracker:
        return 0.0
    
    # Would need to track active order sizes in reward_tracker
    # For now, return 0; can be enhanced with order tracking
    return 0.0
```
