```python
# Combinatorial logical arbitrage: semantic rule-based arbs across related markets
LOGICAL_ARB_ENABLED = _env_bool("LOGICAL_ARB_ENABLED", "false")
LOGICAL_ARB_PRICE_THRESHOLD = _env_float("LOGICAL_ARB_PRICE_THRESHOLD", "0.05")
LOGICAL_ARB_MAX_TRADE_SIZE = _env_float("LOGICAL_ARB_MAX_TRADE_SIZE", "20.0")

# Load logical arb rules from env var (JSON string) or file (logical_arb_rules.json)
LOGICAL_ARB_RULES = []
if LOGICAL_ARB_ENABLED:
    rules_str = os.getenv("LOGICAL_ARB_RULES", "")
    if rules_str:
        try:
            LOGICAL_ARB_RULES = json.loads(rules_str)
        except json.JSONDecodeError as e:
            raise ConfigError(f"Invalid LOGICAL_ARB_RULES JSON: {e}")
    else:
        # Fallback to file
        rules_file = "logical_arb_rules.json"
        if os.path.exists(rules_file):
            with open(rules_file) as f:
                LOGICAL_ARB_RULES = json.load(f)
        else:
            logger.warning("LOGICAL_ARB_ENABLED but no rules provided; feature disabled")
            LOGICAL_ARB_ENABLED = False
```

```python
whale_wallets_str = os.getenv("WHALE_WALLETS", "").strip()
if whale_wallets_str:
    WHALE_WALLETS = [w.strip() for w in whale_wallets_str.split(",") if w.strip()]
if WHALE_COPY_ENABLED and not WHALE_WALLETS:

    WHALE_COPY_ENABLED = False
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI arguments (--mode) → execution path | Untrusted user input; could select invalid modes or trigger unintended strategies |
| Config feature flags → code paths | Feature flags control critical execution paths; incorrect state could cause unintended behavior |
| Dashboard JSON → frontend rendering | Malformed JSON could crash dashboard; stale metrics could mislead trading decisions |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-13 | Tampering | CLI mode selection | mitigate | Argparse enforces choices list; invalid modes rejected at parse time (no bypass possible) |
| T-09-14 | Denial of Service | Invalid LOGICAL_ARB_RULES JSON | mitigate | json.loads() wrapped in try/except; ConfigError raised on decode failure; feature disabled gracefully |
| T-09-15 | Information Disclosure | Dashboard exposes all strategy metrics | accept | Metrics are internal performance data, no sensitive information; bot runs locally or behind auth firewall |
| T-09-16 | Denial of Service | Whale wallet list parsing error | mitigate | WHALE_WALLETS split/trim is fail-safe; invalid addresses logged and skipped (graceful degradation) |
| T-09-17 | Repudiation | Dashboard metric accuracy | mitigate | Metrics queried directly from trades.db (immutable audit log); no in-memory counters that could be manipulated |
| T-09-18 | Denial of Service | HTML template rendering error | mitigate | Jinja2 auto-escaping prevents XSS; template syntax validated at startup (no eval of user input) |
